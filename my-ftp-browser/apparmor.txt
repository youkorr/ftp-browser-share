  # Capacités de sécurité
  deny mount,
  deny umount,
  
  # Autoriser les accès nécessaires
  /data/** rw,
  /config/** rw,
  /ssl/** r,
  
  # Binaires système essentiels
  /bin/* ix,
  /sbin/* ix,
  /usr/bin/* ix,
  /usr/sbin/* ix,
  /init ix,
  
  # Binaires Python
  /usr/bin/python3 ix,
  
  # Accès aux librairies système
  /lib/** rm,
  /usr/lib/** rm,
  /lib64/** rm,
  
  # Accès aux fichiers temporaires
  /tmp/** rwk,
  
  # Accès aux fichiers proc
  /proc/** r,
  /proc/*/attr/** rw,
  
  # Accès réseau
  network inet tcp,
  network inet udp,
}
