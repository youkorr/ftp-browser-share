#include <tunables/global>

profile ftp-browser-share flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  capability,
  file,
  signal,
  mount,
  umount,
  remount,

  # Réseau
  network inet,
  network inet6,
  network tcp,
  network udp,
  network dgram,
  network stream,
  network unix dgram,

  capability setgid,
  capability setuid,
  capability sys_admin,
  capability dac_read_search,

  # S6 Overlay (essentiel pour que /init fonctionne)
  /init rix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /run/{,**} rwk,
  /dev/tty rw,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/sbin/** ix,
  /usr/lib/** r,
  /usr/lib/bashio/** ix,
  /etc/s6/** rix,
  /run/s6/** rix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,

  /var/run/** mrwkl,
  /var/run/ mrwkl,
  /dev/* mrwkl,
  /tmp/** mrkwl,

  # Accès aux fichiers de données
  /data/** rw,
  /config/** rw,

  # Supprimer les warnings `ptrace` avec Docker
  ptrace (trace,read) peer=docker-default,

  # Signal autorisé vers le démon Docker (si nécessaire)
  signal (receive) set=(kill,term) peer=/usr/bin/docker,
}
